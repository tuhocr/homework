Các lệnh về biểu thức chính quy

Từ khóa: regular expression; regex

Escape character

Dấu . sẽ giúp dò ra tất cả các từ. Muốn chọn dấu chấm ký tự trong chữ ID.1 thì ta phải escape dấu . này bằng lệnh \., tuy nhiên trong R thì dấu \ này có 1 ý nghĩa khác, nên ta phải escape thêm \ để trả về được dấu \, nên sau cùng ta có \\. là để định vị dấu . ký tự.

a <- c("lap trinh", "R", "Programming", "study", "ID.1")

a
[1] "lap trinh"   "R"           "Programming" "study"       "ID.1"       
grep(pattern = ".",
     x = a,
     value = T,
     perl = T)
[1] "lap trinh"   "R"           "Programming" "study"       "ID.1"       
grep(pattern = "\\.",
     x = a,
     value = T,
     perl = T)
[1] "ID.1"

Dò con số trong chuỗi

b <- c("study", "ID.1", "1112", "a12", "0", "code663", "id")

b
[1] "study"   "ID.1"    "1112"    "a12"     "0"       "code663" "id"     
grep(pattern = "d",
     x = b,
     value = T,
     perl = T)
[1] "study"   "code663" "id"     
# string chứa digit
grep(pattern = "\\d",
     x = b,
     value = T,
     perl = T)
[1] "ID.1"    "1112"    "a12"     "0"       "code663"
# string không phải là digit
grep(pattern = "\\D",
     x = b,
     value = T,
     perl = T)
[1] "study"   "ID.1"    "a12"     "code663" "id"

Tìm cụm từ trong chuỗi

Cách học các quy tắc tìm và thay thế chuỗi bằng cú pháp regex là ta sẽ thử nghiệm các tham số khác nhau trong lệnh grep() và đánh giá kết quả thu được để hiểu cơ chế lệnh của R, sau đó áp dụng vào tình huống thử nghiệm trên vài chục ký tự trước khi áp dụng vào dữ liệu thực tế.

Nếu việc tìm kiếm phức tạp, ta tách thành nhiều bước nhỏ để lọc dần dữ liệu cho đến khi làm sạch được data đến mức đạt yêu cầu đề bài.

d <- c("happy day",
       "going to school",
       "Working and feel happy everyday",
       "Thanks Happy Birthday",
       "I am hhappy",
       "Today is Monday")

d
[1] "happy day"                       "going to school"                 "Working and feel happy everyday" "Thanks Happy Birthday"           "I am hhappy"                    
[6] "Today is Monday"                
# lấy đúng chữ happy `\\b`
grep(pattern = "\\bhappy",
     x = d,
     perl = T,
     value = T)
[1] "happy day"                       "Working and feel happy everyday"
grep(pattern = "\\b(h|H)appy",
     x = d,
     perl = T,
     value = T)
[1] "happy day"                       "Working and feel happy everyday" "Thanks Happy Birthday"          
# not-word boundary `\\B` lấy chữ happy có những chữ khác
grep(pattern = "\\Bhappy",
     x = d,
     perl = T,
     value = T)
[1] "I am hhappy"
grep(pattern = "happy",
     x = d,
     ignore.case = T,
     perl = T,
     value = T)
[1] "happy day"                       "Working and feel happy everyday" "Thanks Happy Birthday"           "I am hhappy"                    
grep(pattern = "happy",
     x = d,
     ignore.case = F,
     perl = T,
     value = T)
[1] "happy day"                       "Working and feel happy everyday" "I am hhappy"                    
grep(pattern = "happy",
     x = d,
     fixed = T,
     ignore.case = F,
     perl = F,
     value = T)
[1] "happy day"                       "Working and feel happy everyday" "I am hhappy"                    
grep(pattern = "happy",
     x = d,
     fixed = F,
     ignore.case = F,
     perl = T,
     value = T)
[1] "happy day"                       "Working and feel happy everyday" "I am hhappy"                    
grep(pattern = "happy",
     x = d,
     fixed = F,
     ignore.case = F,
     perl = T,
     value = T)
[1] "happy day"                       "Working and feel happy everyday" "I am hhappy"

Thay thế cụm từ

g <- c("ID30", "ID12", "ID9", "id11", "id32", "ID_19")

g
[1] "ID30"  "ID12"  "ID9"   "id11"  "id32"  "ID_19"
# Thay toàn bộ chữ ID (dù viết hoa hay viết thường) 
# theo định dạng CODE_number

gsub(pattern = "ID",
     replacement = "CODE_",
     x = g,
     perl = T)
[1] "CODE_30"  "CODE_12"  "CODE_9"   "id11"     "id32"     "CODE__19"
gsub(pattern = "(ID|id|ID_)",
     replacement = "CODE_",
     x = g,
     perl = T)
[1] "CODE_30"  "CODE_12"  "CODE_9"   "CODE_11"  "CODE_32"  "CODE__19"
# thứ tự ảnh hưởng đến lệnh gsub
gsub(pattern = "(ID_|ID|id)",
     replacement = "CODE_",
     x = g,
     perl = T) -> g_ok

g_ok
[1] "CODE_30" "CODE_12" "CODE_9"  "CODE_11" "CODE_32" "CODE_19"

Ta có thể tách thành 2 bước

g <- c("ID30", "ID12", "ID9", "id11", "id32", "ID_19")

g
[1] "ID30"  "ID12"  "ID9"   "id11"  "id32"  "ID_19"
gsub(pattern = "ID|id",
     replacement = "CODE_",
     x = g,
     perl = T) -> g1

g1
[1] "CODE_30"  "CODE_12"  "CODE_9"   "CODE_11"  "CODE_32"  "CODE__19"
gsub(pattern = "__",
     replacement = "_",
     x = g1,
     perl = T) -> g2

g2
[1] "CODE_30" "CODE_12" "CODE_9"  "CODE_11" "CODE_32" "CODE_19"
identical(g_ok, g2)
[1] TRUE
